function textroll(text) {
  let total = 0;
  let words = text.split(' ').reduce((acc, word) => {
    let letters = word.split('')
    letters.length && acc.push([letters.length, seq(...letters).inhabit(chars)])
    //!letters.length && acc.push([1,silence])
    acc.push([1, silence]);
    total+=letters.length+1;
    return acc;
  },[])
  return timeCat(...words).slow(total)
}
